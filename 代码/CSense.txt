    static parseRule(str) {
      let state = 0;
      let key = "";
      const res = [];
      for (const c of str) {
        if (state === 0) {
          if (c === "[") {
            res.push(key);
            key = "";
            state = 1;
          } else {
            key += c;
          }
        } else if (state === 1) {
          if (c === "]") {
            state = 0;
            res.push({ name: key });
            key = "";
          } else {
            key += c;
          }
        }
      }
      if (state !== 0) throw new Error("Brace unclosed");
      if (key !== "") res.push(key);
      return res;
    }
